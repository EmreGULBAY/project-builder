import express, { Request, Response } from "express";

export const createServer = () => {
    const app = express();

    app.get("/", (req: Request, res: Response) => {
        res.send("success");
    });

    app.get("*", (req: Request, res: Response) => {
        res.status(404).send("Not Found");
    });

    return app;
}
